def execute
  function = Proc.new do
    if File.exist?(@path)
      @data = File.read(@path)
    end
    File.delete(@path)
  end
  super(function)
end

def undo
  function = Proc.new do
    if @data != nil
      f = File.open(@path, 'w')
      f.write @data
      f.close
    end
  end
  super(function)
end
